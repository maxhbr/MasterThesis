\section{Meromorphe Zusammenhängen und Systeme}
\begin{frame}{Meromorphe Zusammenhänge}
  \begin{defn}
    Ein \emph{(Keim eines) Meromorphen Zusammenhangs} ist ein Tupel
    $(\cM,\nabla)$\uncover<2->{, wobei}
    \begin{itemize}
    \item<2-> $\cM$ ist ein $\C(\!\{t\}\!)$-Vektorraum von Dimension $n$
      \uncover<3->{und}
    \item<3-> $\nabla:\cM\to \cM$, welcher die \emph{Leibniz Regel}
      \[
        \nabla(fm)=\frac{d}{dt} f \cdot m + f \nabla(m)
      \]
      für alle $f\in\C(\!\{t\}\!)$ und $m\in\cM$ erfüllt.
    \end{itemize}
  \end{defn}
\end{frame}
\begin{frame}{Systeme und die Gauge Transformation}
  Wähle eine $\C(\!\{t\}\!)$-Basis $\underline{e}=(e_1,e_2,\dots,e_n)$ von
  $\cM$.
  \uncover<2->{%
    \begin{defn}
      \begin{enumerate}
      \item Die \emph{Zusammenhangs-Matrix (von \boldmath$(\cM,\nabla)$ zu
          $\underline{e}$)} ist die Matrix $A\in\Gl_n(\C(\!\{t\}\!))$ so dass
        für jedes $x=\underline{e}\cdot X$
        \[
          \nabla\left(\underline{e}\cdot X\right)=\underline{e}\left(dX-AX\right)
        \]
        erfüllt ist.
      \item<3-> Das \emph{System \boldmath$[A]$} bezeichnet die
        Differentialgleichung $dX=AX$.
      \end{enumerate}
    \end{defn}
  }


  % Sei $A=(a_{jk})_{j,k\in\{1,\dots,n\}}\in\Gl_n(\C(\!\{t\}\!))$ so dass
  % \[
  %   \nabla e_k=-\!\!\sum_{1\leq j\leq n}\!\! a_{jk}(t)e_j
  %   \text{, für alle $k\in\{1,\dots,n\}$.}
  % \]
  % \uncover<2->{%
  % Die Matrix $A$ wird \emph{Zusammenhangs-Matrix (von
  % \boldmath$(\cM,\nabla)$)} genannt und für ein $x=\underline{e}\cdot X$
  % erfüllt sie
  % $\nabla\left(\underline{e}\cdot X\right)=\underline{e}\left(dX-AX\right)$.
  % }
  \uncover<4->{%
    \begin{prop}
      Durch einen Wechsel $F\in\Gl_n\left(\C(\!\{t\}\!)\right)$ der Basis erhält
      man
      das System $\bigl[{}^F\!A\bigr]$, gegeben durch die
      \emph{Gauge-Transformation}:
      \[
        {}^F\!A=(dF)F^{-1} + FAF^{-1} \,.
      \]
    \end{prop}
  }
\end{frame}

\subsection{Levelt-Turittin und die Normalform}
\begin{frame}{Die Normalform}
  Aus dem Levelt-Turittin Theorem erhalten wir:
  \begin{prop}
    Jedes (unverzweigte) System $[A]$ ist formal äquivalent zu einem
    System $[A^0]$ wobei
    \[
      A^0=Q'(t^{-1})+\frac{1}{t}L \,,
    \]
    und
    \begin{itemize}
    \item $Q(t^{-1}):=\bigoplus_{j=1}^sq_j(t^{-1})\cdot\id_{n_j}$ und
    \item $L:=\bigoplus_{j=1}^sL_j$ mit $L_j$ in Jordan-Normalform.
    \end{itemize}
  \end{prop}
  \uncover<2->{
    \begin{defn}
      Jedes System $\tilde{A^0}$ das äquivalent zu einem System $A^0$ wie oben ist,
      nennen wir \emph{normal Form}.
    \end{defn}
  }
\end{frame}

\section{Die Stokes Matrizen}
\begin{frame}{Die Stokes Keime}
  \begin{defn}
    Ein Keim $F\in\Gl_n(\cA_\theta)$ in Richtung $\theta\in S^1$ ist
    \begin{itemize}
    \item ein \emph{Keim der Stokes Garbe \boldmath$\Lambda(A^0)$}, falls er
      \begin{enumerate}
      \item<1-> \emph{multiplikativ flach} ($F$ ist asymptotisch zu $\id$ bei
        $\theta$) \uncover<2->{und}
      \item<2-> eine \emph{Isotropie von \boldmath$A^0$} (${}^FA^0=A^0$)
      \end{enumerate}
      ist.
    \item<3-> Falls er weiter auch noch
      \begin{enumerate}
        \setcounter{enumi}{2}
      \item von \emph{\textcolor{purple}{maximal decay}} \TODO[siehe unten]
      \end{enumerate}
      ist, ist $F$ ein \emph{Stokes Keim}.
      \uncover<4->{
        Die Gruppe dieser Keime wird mit $\Sto_\theta(A^0)$ bezeichnet.
      }
    \end{itemize}
  \end{defn}
\end{frame}
\begin{frame}{\textcolor{purple}{Maximal decay}}
  \begin{defn}
    \begin{enumerate}
    \item<1-> Eine Funktion $e^{q(t^{-1})}$ mit
      $q(t^{-1})\in\frac{a}{t^{k}}+o(t^{-k})$ hat
      \emph{\textcolor{purple}{maximal decay} in Richtung \boldmath$\theta$}
      genau dann wenn $ae^{-ik\theta}$ reell negativ ist.
    \item<2-> Definiere die Relation $q_j\myrel{\theta}q_l$ als äquivalent zu
      \begin{einr}
        $e^{(q_l-q_j)(t^{-1})}$ ist von \textcolor{purple}{maximal decay} in
        Richtung $\theta$.
      \end{einr}
    \end{enumerate}
  \end{defn}
\end{frame}
\begin{frame}{Wann ist $F\in\Lambda_\theta(A^0)$ von \textcolor{purple}{maximal decay}?}
  Ein Keim $F\in\Gl_n(\cA_\theta)$, der die zweite
  Eigenschaft\only<-2>{\footnote{Isotropie von $A^0$.}} erfüllt, sieht wie folgt
  aus
  \begin{align*}
    F  &=t^L e^{Q(t^{-1})}
         \only<1>{
         \underset{\tikzmarkt{e1}}{
         \textcolor{blue!50!white}{\underbrace{\textcolor{black}{
         \left(1_n+\sum_{(l,j)}C^{(l,j)}\right)
         }}}}}
         \only<2->{
         \left(1_n+\sum_{(l,j)}C^{(l,j)}\right)
         }
         e^{-Q(t^{-1})}t^{-L}
    \\ &\onslide<2->{=
         t^L\left(
         1_n+\sum_{(l,j)}C^{(l,j)}e^{(q_l-q_j)(t^{-1})}
         \right)t^{-L} \,.}
  \end{align*}
  \uncover<1>{%
    \begin{flushright}
      \tikzmarkc{n1}{blue} konstant, also in $\Gl_n(\C)$
      \begin{tikzpicture}[remember picture,overlay]
        \draw[->,blue!50!white,thick] (n1) to[out=180,in=270] (e1);
      \end{tikzpicture}
    \end{flushright}
  }
  \uncover<3->{%
    Also ist $F$
    \begin{enumerate}
    \item multiplikativ flach falls falls für jedes $C^{(l,j)}\neq0$ der Faktor
      $e^{(q_l-q_j)(t^{-1})}$ asymptotisch zu $0$ ist \uncover<4->{und}
    \item<4-> von \textcolor{purple}{maximal decay} falls für jedes
      $C^{(l,j)}\neq0$ die Relation $q_j\myrel{\theta}q_l$ erfüllt ist.
    \end{enumerate}
  }
\end{frame}
